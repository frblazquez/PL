int sum([[int, 3], 2] x){
	int row; int col;
	row = 0;
	int suma = 0;
	while (row < 2)
	{
		col = 0;
		while (col < 3)
		{
			suma = suma + x[row][col];
			col = col + 1;
		}
		row = row + 1;
	}
	return suma;
}
