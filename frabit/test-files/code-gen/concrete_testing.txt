void intercambiar(*int v, int i, int j)
{
    v[i] = v[i] + v[j];
    v[j] = v[i] - v[j];
    v[i] = v[i] - v[j];
}

int busqbin(*int v, int clave, int iz, int de)
{
    int m = (iz + de) / 2;
    int retval;
    if (v[m] == clave)
    {
        retval = m;
    }
    else
    {
        if (v[m] < clave)
        {
            retval = busqbin(v,clave,m + 1,de);
        }
        else { retval = busqbin(v,clave,iz,m - 1); }
    }
    return retval;
}

void bubblesort(*int v, int iz, int de)
{
    bool swapped = true;
    while (swapped)
    {
        swapped = false;
        int i = iz + 1;
        while (i <= de)
        {
            if (v[i - 1] > v[i])
            {
                intercambiar(v,i - 1,i);
                swapped = true;
            }
            i = i + 1;
        }
    }
}
