void intercambiar(*int v, int i, int j)
{
    v[i] = v[i] + v[j];
    v[j] = v[i] - v[j];
    v[i] = v[i] - v[j];
}



void bubblesort(*int v, int iz, int de)
{
    bool swapped = true;
    while (swapped)
    {
        swapped = false;
        int i = iz + 1;
        while (i <= de)
        {
            if (v[i - 1] > v[i])
            {
                intercambiar(v,i - 1,i);
                swapped = true;
            }
            i = i + 1;
        }
    }
}
